import fs from 'fs'
import path from 'path'
import { fileURLToPath } from 'url'

import { withProgressTyping } from '../../behavior/typing.js'

const __filename = fileURLToPath(import.meta.url)
const __dirname = path.dirname(__filename)

export const flowLogin = addKeyword(EVENTS.WELCOME).addAnswer(
  messageWelcome,

  async (ctx, { provider, flowDynamic, fallBack }) => {
    const reqId = `${Date.now().toString(36)}-${Math.random().toString(36).slice(2, 7)}`

    const isValid = validateCode(code_student)
    console.log(`${TAG}(${reqId}) validateCode â†’ ${isValid}`)
    if (!isValid) {
      console.log(`${TAG}(${reqId}) fallBack (cÃ³digo invÃ¡lido)`)
      return fallBack('âŒ CÃ³digo no vÃ¡lido. Ingresa un cÃ³digo correcto (Ej: U202012345): ')

        'Validando cÃ³digo UPCâ€¦ ðŸ”Ž',
        // workFn: NO envÃ­a mensajes; solo retorna data
        async () => {
          const t0 = Date.now()
          console.log(`${TAG}(${reqId}) axios.request â†’ POST ${GAS_URL} body={ userCode: "${code_student}" }`)
          try {
            const res = await axios.post(
              GAS_URL,
              { userCode: code_student },
              {
                timeout: 8000,
                // asegÃºrate de que trate no-2xx como error:
                validateStatus: (s) => s >= 200 && s < 300,
                headers: { 'Accept': 'application/json', 'Cache-Control': 'no-cache' },
                transformResponse: [(data) => {
                  // Intenta parsear JSON si viene como string
                  try { return typeof data === 'string' ? JSON.parse(data) : data }
                  catch { return { raw: data } }
                }]
              }
            )

            console.log(`${TAG}(${reqId}) axios.response â† status=${res.status} elapsed=${dt}ms data=`, res.data)
            return res.data
          } catch (err) {
            const dt = Date.now() - t0
            // Detalle de Axios
            if (axios.isAxiosError(err)) {
              const s = err.response?.status
              const d = err.response?.data
              console.error(`${TAG}(${reqId}) axios.error Ã— status=${s ?? 'N/A'} elapsed=${dt}ms message="${err.message}" data=`, d)
            } else {
              console.error(`${TAG}(${reqId}) request.error Ã— elapsed=${dt}ms`, err)

          }
        },
        // onSuccess: aquÃ­ sÃ­ respondes al usuario
        async (data) => {
          console.log(`${TAG}(${reqId}) onSuccess â†’ data=`, data)
          if (!data || data.raw) {
            console.warn(`${TAG}(${reqId}) respuesta inesperada (no JSON vÃ¡lido).`)

        },

  }

// Validador: U + 9 alfanumÃ©ricos (normaliza y limpia separadores)
function validateCode(codigo) {
  const norm = String(codigo)
    .normalize('NFKD')
    .toUpperCase()
    .replace(/[^A-Z0-9]/g, '')
  const ok = /^U[A-Z0-9]{9}$/.test(norm)
  // Log especÃ­fico del validador
  console.log(`${TAG} validateCode â†’ input="${codigo}" norm="${norm}" ok=${ok}`)
  return ok
}
